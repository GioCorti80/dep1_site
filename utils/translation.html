import pyspark
from pyspark.sql import SparkSession

# 1. Initialize Spark Session
spark = SparkSession.builder.appName("VersionCheck").getOrCreate()

print("--- CLUSTER VERSION DIAGNOSTICS ---")

# 2. Check PySpark and Spark Versions
print(f"PySpark Version: {pyspark.__version__}")
print(f"Spark Engine Version: {spark.version}")

# 3. Check Scala Version (Extracting from the JVM)
try:
    scala_version = spark.sparkContext._jvm.scala.util.Properties.versionString()
    print(f"Scala Version: {scala_version}")
except Exception as e:
    print(f"Could not determine Scala version: {e}")

# 4. Check User-Injected JARs (This verifies your Airflow config worked)
try:
    user_jars = spark.sparkContext.getConf().get("spark.jars")
    print(f"Custom JARs passed to Spark: {user_jars}")
except Exception:
    print("No custom JARs found in spark.jars")

# 5. Find the active BigQuery Connector in the JVM Classpath
try:
    # This pulls every library currently loaded into the Spark driver's memory
    jvm_classpath = spark.sparkContext._jvm.java.lang.System.getProperty("java.class.path")
    
    # Filter only the BigQuery related JARs
    bq_jars = [jar for jar in jvm_classpath.split(':') if 'bigquery' in jar.lower()]
    
    print("Loaded BigQuery Connector JARs:")
    for jar in bq_jars:
        print(f" -> {jar}")
except Exception as e:
    print(f"Could not inspect JVM classpath: {e}")

print("-----------------------------------")
