# 2. Use BigQuery's native STRING_AGG(DISTINCT ...) to group and concatenate

    SELECT 
        NDG, 
        NDG_TITOLARE_COLLEGATO, 
        STATO_RAPPORTO_COLLEGATO, 
        SERVIZIO_COLLEGATO, 
        FILIALE_COLLEGATO, 
        CONTO_COLLEGATO, 
        DT_ACCENSIONE_RAP_COLL, 
        DT_VARIAZIONE_RAP_COLL,
        
        -- The equivalent of concat_ws(" ", collect_set(...))
        STRING_AGG(DISTINCT TIPO_COLLEGAMENTO, ' ') AS COD_TIPO_LEGAME,
        STRING_AGG(DISTINCT DESC_TIPO_COLLEGAMENTO, ' ') AS DESC_TIPO_LEGAME,
        STRING_AGG(DISTINCT DESC_COD_TIPO_LEGAME, ' ') AS DESC_COD_TIPO_LEGAME
        
    FROM `{temp_table_id}`
    GROUP BY 1, 2, 3, 4, 5, 6, 7, 8
"""
