# 1. Strip any accidental whitespace
clean_str = trd1a_filt['VALORE_DATO_NDG'].str.strip()

# 2. Check if it is EXACTLY 8 digits (filters out letters, slashes, short/long strings)
is_eight_digits = clean_str.str.contains(r'^\d{8}$').fillna(False)

# 3. Keep only the 8-digit strings, turn the rest to None
clean_str = clean_str.where(is_eight_digits, None)

# 4. Filter out the fake "zero" dates we found earlier
clean_str = clean_str.replace({
    "00000000": None, 
    "0000000": None
})

# 5. Now it should be safe to parse!
trd1a_filt['DT_SCADENZA_DOC'] = bpd.to_datetime(clean_str, format='%d%m%Y')
