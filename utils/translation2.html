def read_bq(spark, query):
    return (spark.read.format("bigquery")
            .option("materializationProject", "your-project-id")
            .option("materializationDataset", "your_temp_dataset")
            .option("viewsEnabled", "true")
            .option("query", query)
            .load())

            .option("readDataFormat", "ARROW") 
      .option("useAvroLogicalTypes", "true")

def read_bq_as_parquet(spark, table, bucket, path):
    df = spark.read.format("bigquery") \
        .option("readMethod", "export") \
        .option("temporaryGcsBucket", bucket) \
        .load(table)

    df.write.mode("overwrite").parquet(path)
    return spark.read.parquet(path)
